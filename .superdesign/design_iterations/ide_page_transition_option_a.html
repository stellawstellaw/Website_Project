// OPTION A: Fade + Slide Overlay
// Pink overlay slides up while fading in, then slides away

function pageTransition(callback) {
  const overlay = document.querySelector('.page-transition');
  const tl = gsap.timeline();

  // Slide up from bottom while fading in
  tl.fromTo(overlay, {
    opacity: 0,
    y: '100%'
  }, {
    opacity: 1,
    y: '0%',
    duration: 0.4,
    ease: 'power2.inOut'
  })

  // Callback to change page content
  .call(callback)

  // Slide up and fade out
  .to(overlay, {
    opacity: 0,
    y: '-100%',
    duration: 0.4,
    ease: 'power2.inOut',
    delay: 0.1
  });
}
