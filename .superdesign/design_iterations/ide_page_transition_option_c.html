// OPTION C: Split Curtains
// Two panels slide in from sides, then slide away

function pageTransition(callback) {
  const left = document.querySelector('.curtain-left');
  const right = document.querySelector('.curtain-right');
  const tl = gsap.timeline();

  // Slide curtains in from sides
  tl.to(left, {
    left: '0%',
    duration: 0.4,
    ease: 'power2.inOut'
  })
  .to(right, {
    right: '0%',
    duration: 0.4,
    ease: 'power2.inOut'
  }, '<') // Same time as left

  // Change page content
  .call(callback)

  // Slide curtains away
  .to(left, {
    left: '-50%',
    duration: 0.4,
    ease: 'power2.inOut',
    delay: 0.1
  })
  .to(right, {
    right: '-50%',
    duration: 0.4,
    ease: 'power2.inOut'
  }, '<');
}
