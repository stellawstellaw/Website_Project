// OPTION A: Subtle Scale + Shadow
element.addEventListener('mouseenter', () => {
  gsap.to(element, {
    scale: 1.05,
    boxShadow: '0 10px 30px rgba(0,0,0,0.15)',
    duration: 0.3,
    ease: 'power2.out'
  });
});
element.addEventListener('mouseleave', () => {
  gsap.to(element, {
    scale: 1,
    boxShadow: '0 4px 15px rgba(0,0,0,0.05)',
    duration: 0.3,
    ease: 'power2.out'
  });
});

  });
