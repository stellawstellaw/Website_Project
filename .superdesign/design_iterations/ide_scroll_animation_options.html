// OPTION C: Slide From Sides (alternating)
gsap.from('.section-header', {
  scrollTrigger: { trigger: '.highlights-section', start: 'top 80%' },
  y: -30, opacity: 0, duration: 0.5, ease: 'power2.out'
});

// Cards from left
gsap.from('.highlight-card:nth-child(odd)', {
  scrollTrigger: { trigger: '.highlights-section', start: 'top 70%' },
  x: -100, opacity: 0, stagger: 0.2, duration: 0.7, ease: 'power3.out'
});

// Cards from right
gsap.from('.highlight-card:nth-child(even)', {
  scrollTrigger: { trigger: '.highlights-section', start: 'top 70%' },
  x: 100, opacity: 0, stagger: 0.2, duration: 0.7, ease: 'power3.out'
});
