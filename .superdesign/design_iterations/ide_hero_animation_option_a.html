// OPTION A: Dramatic Stagger - Bold, playful entrance

const heroTl = gsap.timeline();

// Title lines: Large Y offset, slow stagger, elastic ease
heroTl.from('.hero-text h1 span', {
  y: 100,
  opacity: 0,
  stagger: 0.3,
  duration: 1,
  ease: 'elastic.out(1, 0.5)'
})

// Subtitle fades in
.from('.hero-text p', {
  opacity: 0,
  y: 20,
  duration: 0.6,
  ease: 'power2.out'
}, '-=0.3')

// Buttons pop in together
.from('.btn', {
  opacity: 0,
  scale: 0.8,
  stagger: 0.1,
  duration: 0.5,
  ease: 'back.out(1.7)'
}, '-=0.2')

// Mini-cards cascade from left
.from('.mini-card', {
  x: -200,
  opacity: 0,
  stagger: 0.15,
  duration: 0.8,
  ease: 'power3.out'
}, '-=0.3');
