// OPTION B: Expanding Circle
// Circle expands from center, then contracts away

function pageTransition(callback) {
  const circle = document.querySelector('.page-transition');
  const maxSize = Math.max(window.innerWidth, window.innerHeight) * 2.5;
  const tl = gsap.timeline();

  // Expand circle from center
  tl.to(circle, {
    width: maxSize,
    height: maxSize,
    duration: 0.5,
    ease: 'power2.in'
  })

  // Change page content
  .call(callback)

  // Contract circle away
  .to(circle, {
    width: 0,
    height: 0,
    duration: 0.5,
    ease: 'power2.out',
    delay: 0.1
  });
}
