// OPTION B: Subtle Elegance - Clean, professional entrance

const heroTl = gsap.timeline();

// Title lines: Small Y offset, fast stagger, power ease
heroTl.from('.hero-text h1 span', {
  y: 30,
  opacity: 0,
  stagger: 0.15,
  duration: 0.6,
  ease: 'power3.out'
})

// Subtitle fades in smoothly
.from('.hero-text p', {
  opacity: 0,
  duration: 0.5,
  ease: 'power2.out'
}, '-=0.2')

// Buttons fade in together
.from('.btn', {
  opacity: 0,
  y: 15,
  stagger: 0.1,
  duration: 0.4,
  ease: 'power2.out'
}, '-=0.2')

// Mini-cards fade up gently
.from('.mini-card', {
  y: 40,
  opacity: 0,
  stagger: 0.1,
  duration: 0.5,
  ease: 'power2.out'
}, '-=0.2');
